LibraryManagementSystem Low Level Design UML Diagram

Functional Requirements
1. User can login/signup
2. Librarian can add/remove books
3. User can search for books by title, author, or ISBN
4. User can borrow/return books
5. Librarian can manage book inventory
6. User can view borrowed book history
7. Librarian can view user borrowing history
8. User can pay fines for late returns
9. Librarian can generate reports on book usage and user activity
10. System should send notifications for due dates and overdue books



classes:-
1. User
2. Librarian
3. Books
4. Borrowing
5. Payment
6. Notification
7. Report

class User {
    int userID;
    string userName;
    string userEmail;
    string password;
    string userAddress;

    void login(string userEmail, string password);
    void signup(string userName, string userEmail, string password, string userAddress);
    void searchBook(string title, string author, string ISBN);
    void borrowBook(int bookID);
    void returnBook(int bookID);
    void viewBorrowedHistory();
    void payFine(double amount);
};
class Librarian {
    int librarianID;
    string librarianName;
    string librarianEmail;
    string password;

    void login(string librarianEmail, string password);
    void addBook(Book book);
    void removeBook(int bookID);
    void manageInventory(int bookID, int quantity);
    void viewUserBorrowingHistory(int userID);
    void generateReport(string reportType);
};
class Book {
    int bookID;
    string title;
    string author;
    string ISBN;
    string category; // e.g., Fiction, Non-Fiction, Science
    double price;
    bool isAvailable;
    int stockQuantity;
    void updateStock(int quantity);
    bool checkAvailability();
};
class Borrowing {
    int borrowingID;
    int userID;
    int bookID;
    Date borrowDate;
    Date dueDate;
    Date returnDate;
    double calculateFine();
};
class Payment {
    int paymentID;
    int userID;
    double amount;
    Date paymentDate;
    string paymentMethod; // e.g., Credit Card, PayPal
    string paymentStatus; // e.g., Pending, Completed, Failed
    void processPayment(int userID, double amount, string paymentMethod);
};
class Notification {
    int notificationID;
    int userID;
    string message;
    Date notificationDate;
    void sendNotification(int userID, string message);
};
class Report {
    int reportID;
    string reportType; // e.g., Book Usage, User Activity
    Date generatedDate;
    void generateReport(string reportType);
};
